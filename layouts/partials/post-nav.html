{{- /* Previous/next post navigation, chronological, skipping external (customPermalink) posts. */ -}} {{- if eq .Section "post" -}} {{- $posts := slice -}} {{- range (where .Site.RegularPages "Section" "post").ByDate -}} {{- if not .Params.customPermalink -}}{{- $posts = $posts | append . -}}{{- end -}} {{- end -}} {{- $cur := .RelPermalink -}} {{- $prev := false -}}{{- $next := false -}} {{- range $i, $p := $posts -}} {{- if eq $p.RelPermalink $cur -}} {{- if gt $i 0 -}}{{- $prev = index $posts (sub $i 1) -}}{{- end -}} {{- if lt (add $i 1) (len $posts) -}}{{- $next = index $posts (add $i 1) -}}{{- end -}} {{- end -}} {{- end -}} {{- if or $prev $next -}}
{{- with $prev }} ← Previous {{ .Title }} {{- end }} {{- with $next }} Next → {{ .Title }} {{- end }}
{{- end -}} {{- end -}}
